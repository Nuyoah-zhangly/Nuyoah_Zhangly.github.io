    函数声明方式

    /**
     * 函数声明方式1：直接声明方式
     */
    function func1(a, b, c) {
        console.log(a + b + c);
    }
    func1(2,3,4);//9
 
    /**
     * 函数定义方式二：函数表达式
     */
    var func2 = function (a, b, c) {
        console.log(a + b + c);
    };
    func2(1, 2, 3);//6
 
    /**
     * 函数定义三：通过构造函数方式
     * var 变量名 = new Function('形参1','形参2'....,'函数体');
     * 注意:参数可以有多个
     */
    var func3 = new Function('a', 'b', 'c', 'console.log(a+b+c)');
    func3(2, 23, 2);//27
